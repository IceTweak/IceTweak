package main

import (
	"fmt"
	"sync"
	"unsafe"
)

var secret = [...]byte{72, 101, 108, 108, 111, 44, 32, 87, 111, 114, 108, 100, 33}

type obscure struct {
	ptr *byte
	len int
	done chan struct{}
}

func main() {
	var wg sync.WaitGroup
	wg.Add(1)

	ob := &obscure{
		ptr: &secret[0],
		len: len(secret),
		done: make(chan struct{}),
	}

	go func(o *obscure) {
		<-o.done
		msg := unsafe.Slice(o.ptr, o.len)
		fmt.Println(string(msg))
		wg.Done()
	}(ob)

	close(ob.done)
	wg.Wait()
}
